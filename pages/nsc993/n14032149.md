  <ok href="https://www.epochtimes.com/gb/tag/%E6%9C%9D%E9%B2%9C%E5%8C%96.html">
   朝鲜化
  </ok>

  <ok href="https://www.epochtimes.com/gb/tag/%E7%BB%8F%E6%B5%8E%E5%B4%A9%E7%9B%98.html">
   经济崩盘
